   138	    private fun blockedDiceFor(moves: IntArray, d0: Int, d1: Int): Set<Int> {
   139	        if (d0 == d1) return emptySet()
   140	        val blocked = mutableSetOf<Int>()
   141	        val s0 = (0 until moves.size step 8).any { m ->
   142	            (0 until 8 step 2).any { j -> moves[m+j] >= 0 && moves[m+j] - moves[m+j+1] == d0 } }
   143	        val s1 = (0 until moves.size step 8).any { m ->
   144	            (0 until 8 step 2).any { j -> moves[m+j] >= 0 && moves[m+j] - moves[m+j+1] == d1 } }
   145	        return blocked
   146	    }

   184	        if (_gameState.value.phase != GamePhase.WAITING_FOR_ROLL) return
   185	        viewModelScope.launch(engineThread) {

   199	                val allMoves = Engine.getLegalMoves(board, d0, d1)
   200	                android.util.Log.i("gnubg-vm", "blocked check d0=$d0 d1=$d1 moves=${allMoves.size/8}")

   207	                    engineDice    = null,
   208	                    blockedDice   = blockedDiceFor(allMoves, d0, d1)

   213	                else {
